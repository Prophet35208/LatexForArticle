\begin{frame}{Объединение методов}

\textbf{Два подхода к совместному использованию LLM и TDD:}

\vspace{0.5cm}

\begin{enumerate}
    \item \textbf{LLM для генерации тестов}
    \begin{itemize}
        \item Генерируем тесты по бизнес-требованиям
        \item Нейросеть формализует ожидаемое поведение системы
        \item Тесты становятся исполняемой спецификацией (TDD)
        \item Цель: быстрое создание качественных тестов до написания кода
    \end{itemize}

    \vspace{0.5cm}

    \item \textbf{LLM для написания кода по тестам}
    \begin{itemize}
        \item Человек пишет тесты как формальную спецификацию
        \item Нейросеть генерирует код, проходящий все проверки
        \item Цель: генерация более точного и качественного кода
        \item Тесты служат и промптом, и критерием приёмки
    \end{itemize}
\end{enumerate}

\end{frame}
