\begin{frame}{LLM}
\textbf{Особенности:}
\begin{itemize}
    \item Быстрая генерация кода по промпту
    \item Поддержка множества языков программирования
    \item Способность обрабатывать сложные запросы на естественном языке
\end{itemize}

\vspace{0.8cm}
\textbf{Недостатки:}
\begin{itemize}
    \item Высокая вероятность ошибок, скрытых багов
    \item Неточность модели при работе с большими проектами
\end{itemize}
\end{frame}
